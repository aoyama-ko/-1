<script>

function diagnose(){

let bowel = +document.getElementById("bowel").value;

let bloating = +document.getElementById("bloating").value;

let food = +document.getElementById("food").value;

let sleep = +document.getElementById("sleep").value;

let stress = +document.getElementById("stress").value;

let cold = +document.getElementById("cold").value;


let total = bowel + bloating + food + sleep + stress + cold;

let score = Math.round(total / 30 * 100);


let type = "";

let advice = "";


if(score < 40){

  type = "❄️ 冷え・腸停滞タイプ";

  advice = "腸の動きが低下し、血流不足の可能性があります。腹部を温めるヨガやツイストポーズ、白湯習慣を取り入れましょう。";

}

else if(score < 70){

  type = "🌙 自律神経乱れタイプ";

  advice = "ストレスや睡眠の質が腸に影響している可能性があります。深い呼吸・リラックス系ヨガ・腸もみセルフケアがおすすめです。";

}

else{

  type = "🌿 バランス良好タイプ";

  advice = "腸内環境は比較的良好です。今の生活を維持しつつ、定期的なメンテナンスヨガでさらに安定させましょう。";

}


document.getElementById("output").innerHTML =

"<div class='result'>" +

"<h2>診断結果</h2>" +

"<p><strong>腸活スコア：</strong>" + score + "点</p>" +

"<p><strong>タイプ：</strong>" + type + "</p>" +

"<p><strong>アドバイス：</strong>" + advice + "</p>" +

"</div>";


new Chart(document.getElementById("chart"), {

type: 'radar',

data: {

labels: ["便通","張り","食事","睡眠","ストレス","巡り"],

datasets: [{

label: "あなたの状態",

data: [bowel, bloating, food, sleep, stress, cold]

}]

},

options: {

scales: {

r: {

min: 0,

max: 5

}

}

}

});

}
